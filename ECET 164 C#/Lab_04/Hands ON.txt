string mathType;
        private void listBox1_SelectedIndexChanged(object sender, EventArgs e)
        {
            mathType = listBox1.SelectedItem.ToString();

            switch (mathType)
            {
                case "Add":
                    label1.Text = "add first number:";
                    label2.Text = "add second number:";
                    label3.Text = "+";
                    label4.Text = "=";
                    break;
                case "Subtract":
                    label1.Text = "Enter first number:";
                    label2.Text = "subtract second number:";
                    label3.Text = "-";
                    label4.Text = "=";
                    break;
                case "Multiply":
                    label1.Text = "Multiply first number:";
                    label2.Text = "Multiply second number:";
                    label3.Text = "*";
                    label4.Text = "=";
                    break;
                default:
                    MessageBox.Show("Select an option in the list box");
                    break;

            }
        }
